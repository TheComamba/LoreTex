\newcommand{\setContinent}[1]{
	\directlua{
		CurrentContinent = "#1"
		CurrentRegion = ""
		CurrentCity = ""
	}
}

\newcommand{\setRegion}[1]{
	\directlua{
		CurrentRegion = "#1"
		CurrentCity = ""
	}
}

\newcommand{\setCity}[1]{
	\directlua{
		CurrentCity = "#1"
	}
}

\newcommand{\resetCurrentLabels}{
	\directlua{
		CurrentContinent = ""
		CurrentRegion = ""
		CurrentCity = ""
	}
}

\newcommand{\newEntity}[4]{
	\directlua{
		NewEntity("#1","#2", [[\detokenize{#3}]], [[\detokenize{#4}]])
	}
}

\newcommand{\newContinent}[3]{
	\newEntity{#1}{continent}{#2}{#3}
	\setContinent{#1}
}

\newcommand{\newRegion}[3]{
	\newEntity{#1}{region}{#2}{#3}
	\directlua{SetDescriptor(CurrentEntity(),"parent", CurrentContinent)}
	\setRegion{#1}
}

\newcommand{\newCity}[3]{
	\newEntity{#1}{city}{#2}{#3}
	\directlua{SetDescriptor(CurrentEntity(),"parent", CurrentRegion)}
	\setCity{#1}
}

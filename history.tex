\newcommand{\addHistory}[3][0]{
	\directlua{
		if true then
			local item = EmptyHistoryItem()
			item["originator"] = GetMainLabel(CurrentEntity())
			SetDay(item, "#1")
			SetYear(item, "#2")
			item["event"] = [[\detokenize{#3}]]
			ProcessEvent(item)
		end
	}
}
